Create DATABASE SOCIAL;
use social;

# Tabella Utenti 
CREATE TABLE utenti (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(30) NOT NULL UNIQUE, # Aggiunto username
    nome VARCHAR(50) NOT NULL,
    cognome VARCHAR(50) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    data_nascita DATE,
    bio TEXT,
    foto_profilo VARCHAR(255) DEFAULT 'default.png',
    data_registrazione TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

# Tabella Post
CREATE TABLE post (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_utente INT NOT NULL,
    contenuto TEXT,
    data_post TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_utente) REFERENCES utenti(id) ON DELETE CASCADE
);

# Tabella Media 
CREATE TABLE media_post (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_post INT NOT NULL,
    tipo ENUM('immagine','video') NOT NULL,
    percorso_file VARCHAR(255) NOT NULL,
    FOREIGN KEY (id_post) REFERENCES post(id) ON DELETE CASCADE
);

# Tabella Amicizie 
CREATE TABLE amicizie (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_utente_mandante INT NOT NULL,
    id_utente_ricevente INT NOT NULL,
    stato ENUM('richiesta','accettata','rifiutata') DEFAULT 'richiesta',
    data_richiesta TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE (id_utente_mandante, id_utente_ricevente), # Impedisce doppioni
    FOREIGN KEY (id_utente_mandante) REFERENCES utenti(id) ON DELETE CASCADE,
    FOREIGN KEY (id_utente_ricevente) REFERENCES utenti(id) ON DELETE CASCADE
);

# Tabella Notifiche
CREATE TABLE notifiche (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_utente_destinatario INT NOT NULL,
    id_utente_mittente INT NOT NULL,
    tipo ENUM('like', 'commento', 'amicizia') NOT NULL,
    id_riferimento INT, # ID del post o del commento correlato
    letto BOOLEAN DEFAULT FALSE,
    data_notifica TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_utente_destinatario) REFERENCES utenti(id) ON DELETE CASCADE
);

# Tabella Commenti
CREATE TABLE commenti (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_post INT NOT NULL,
    id_utente INT NOT NULL,
    contenuto TEXT,
    data_commento TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (id_post) REFERENCES post(id) ON DELETE CASCADE,
    FOREIGN KEY (id_utente) REFERENCES utenti(id) ON DELETE CASCADE
);

# Tabella media nei commenti dei post
CREATE TABLE media_commenti (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_commento INT NOT NULL,
    tipo ENUM('immagine','video'),
    percorso_file VARCHAR(255),

    FOREIGN KEY (id_commento) REFERENCES commenti(id) ON DELETE CASCADE
);

# Tabella Likes
CREATE TABLE likes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_post INT NOT NULL,
    id_utente INT NOT NULL,
    data_like TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    UNIQUE (id_post, id_utente),

    FOREIGN KEY (id_post) REFERENCES post(id) ON DELETE CASCADE,
    FOREIGN KEY (id_utente) REFERENCES utenti(id) ON DELETE CASCADE
);

# Tabella Chat
CREATE TABLE chat (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_mittente INT NOT NULL,
    id_destinatario INT NOT NULL,
    messaggio TEXT,
    data_messaggio TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (id_mittente) REFERENCES utenti(id) ON DELETE CASCADE,
    FOREIGN KEY (id_destinatario) REFERENCES utenti(id) ON DELETE CASCADE
);

# Tabella media nella chat
CREATE TABLE media_chat (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_messaggio INT NOT NULL,
    tipo ENUM('immagine','video'),
    percorso_file VARCHAR(255),

    FOREIGN KEY (id_messaggio) REFERENCES messaggi(id) ON DELETE CASCADE
);
